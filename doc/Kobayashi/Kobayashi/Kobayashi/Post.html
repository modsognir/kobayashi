def find(*args)
  # Reformat arguments (one-digit months and days should be converted to two-digit format)
  args = args.map {|a| a.to_s =~ /^\d{1}$/ ? "%02d" % a : a }
  # Look for all files which contain the dashed format ie. 2011/09 looks for 2011-09 in the post files
  posts = post_files.select {|pf| File.basename(pf) =~ /#{args.join('-')}/ }.map {|pf| new(pf) }

end

def parse_post(post)
  # Parse all the datas!!! Splits based on Jakyll markdown format
  marker, meta, body = File.read(post.to_s).split("---\n", 3)
  post = OpenStruct.new YAML.load(meta)
  # yields metadata from Jekyll wp converted posts
  # post.layout, post.title, post.date, post.comments, post.published, post.categories(an array) etc
  post.title = post.title
  post.slug = post.title.sluggify
  post.date = (Time.parse(post.date.to_s)).to_date
  post.body = body
  post.tags = post.categories
  post.url  = "/#{post.date.year}/#{"%02d" % post.date.month}/#{"%02d" % post.date.day}/#{post.slug}"
  post.file = "/#{post.date.year}/#{"-" % post.date.month}/#{"-" % post.date.day}/#{post.slug}"
  post
end
